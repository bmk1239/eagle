# -------------------------------------# -------------------------------------- name: Harvest FreeTV cookies
  id: harvest
  env:
    CHROME_PATH: ${{ steps.chrome.outputs.chrome-path }}
    IL_PROXY   : ${{ env.IL_PROXY }}
  run: |
    node - <<'JS'
      const puppeteer = require('puppeteer-core');
      (async () => {
        const browser = await puppeteer.launch({
          executablePath: process.env.CHROME_PATH,
          headless: 'new',
          args: [
            '--no-sandbox',
            '--disable-setuid-sandbox',
            '--ignore-certificate-errors',          // <â”€ NEW
            `--proxy-server=${process.env.IL_PROXY}`
          ]
        });
        const page = await browser.newPage();
        await page.goto('https://web.freetv.tv/', { waitUntil: 'networkidle2' });

        const cookies = (await page.cookies())
          .map(c => `${c.name}=${c.value}`).join('; ');

        console.log('::add-mask::' + cookies);           // hide
        console.log(`FREETV_COOKIE_HARVESTED=${cookies}`); // export
        await browser.close();
      })();
    JS
